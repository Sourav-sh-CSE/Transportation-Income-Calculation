CREATE TABLE Location (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(30) NOT NULL,
price INT(10)
)

CREATE TABLE Party (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(30) NOT NULL,
location INT(10),
ghor_id INT(10)
)

CREATE TABLE Truck (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
t_num VARCHAR(30) NOT NULL,
mobile VARCHAR(30) NOT NULL,
rent INT(10),
mal INT(10), 
adv INT(10), 
totalTk INT(10),
avgMotiTk INT(10),
motiG INT(10),
t_inc INT(10),
iGet INT(10)  
)

CREATE TABLE Chalan (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
party_id INT(10) NOT NULL,
mal INT(10) NOT NULL,
truck_num VARCHAR(50)     
)

CREATE TABLE Ghor (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(50)     
)

CREATE TABLE Mal (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
party_id INT(10) NOT NULL,
total_mal INT(10) NOT NULL
)
